js切换效果源代码如下：

<script>
    $(document).ready(function(){
        //顶部导航栏的切换效果
        $("#mytab1 li").click(function (e) {
            e.preventDefault();
            $("#mytab1 li").removeClass("active");
            $(this).addClass("active");
        });
        //一般导航栏的切换效果
        $('#mytab2 li').click(function (e) {
            e.preventDefault();
            $("#mytab2 li").removeClass("active");
            $(this).addClass("active");
        });
        $('#mytab3 li').click(function (e) {
            e.preventDefault();
            $("#mytab3 li").removeClass("active");
            $(this).addClass("active");
        });
        $('#mytab4 li').click(function (e) {
            e.preventDefault();
            $("#mytab4 li").removeClass("active");
            $(this).addClass("active");
        });
        $('#mytab5 li').click(function (e) {
            e.preventDefault();
            $("#mytab5 li").removeClass("active");
            $(this).addClass("active");
        });
    });
</script>
